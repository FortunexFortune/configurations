                                                       # On its own it's rarely useful — its real value comes from filtering, 
                                                       # which is why you almost always see it with parameters like -Noun, -Verb, -Module, or a name pattern:

                                                        Get-Command *iis*              # anything with "iis" in the name
                                                        Get-Command -Module SqlServer  # everything from one module
                                                        (Get-Command).Count            # how many commands you have (fun fact check)

Get-ExecutionPolicy                                    # shows PowerShell's current script execution policy — the setting that controls whether PowerShell will run script files (.ps1) on the machine.
                                                        Restricted — no scripts run at all; interactive commands only. (Old default on Windows clients)
                                                        AllSigned — only scripts signed by a trusted publisher run.
                                                        RemoteSigned — locally created scripts run freely; scripts downloaded from the internet must be signed. (Default on Windows Server)
                                                        Unrestricted — everything runs, with a warning prompt for downloaded scripts.
                                                        Bypass — everything runs, no warnings, no prompts.
                                                        Undefined — no policy set at that scope.
