
helm version


helm repo add gitlab https://charts.gitlab.io/        # add repo to helm repostiroy 


helm repo list                                        # list repos 
helm repo update                                      # update all repos
helm search repo                                      # search all charts
helm search repo chart-name                           # search for chart in local repos

helm search repo chart-name --version                # search for chart and theri versions

helm show chart chart-name                            # return the contents of a chart


helm show values chart-name                           # returns values from chart 
helm get values chart-name                            # returns values from installed chart  
helm get values chart-name  --all                     # returns values from installed chart  


helm repo remove name                                 # remove repo



helm install [release name] [chart name]                   # Install specific version
helm install local-wp bintami/wordpress --version=23.1.20  # Install specific version



helm list                                             # return list of installed releases 
helm uninstall   release-name                         # remove release
